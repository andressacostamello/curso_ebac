# Importar biblioteca GitHub
from github import Github

# Informações de autenticação
username = "seu_nome_de_usuario"
password = "sua_senha"

# Autenticar no GitHub
g = Github(username, password)

# Criar um novo repositório
repo_name = "curso_ebac_frontend"
repo = g.get_user().create_repo(repo_name)

# Criar uma nova branch
branch_name = "exercicio_git"
branch = repo.create_git_ref(ref=f"refs/heads/{branch_name}")

# Adicionar arquivo nome.txt à branch
file_content = "Seu Nome Completo"
file_name = "nome.txt"
file_path = f"{branch_name}/{file_name}"
repo.create_file(file_path, "Adicionar nome.txt", file_content, branch=branch_name)

# Obter link do repositório
repo_url = repo.html_url

print("Repositório criado com sucesso!")
print("Link do repositório:", repo_url)
